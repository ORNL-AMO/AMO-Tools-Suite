17  }
18  else if (lineFreq_ == Motor::LineFrequency::FREQ50){
19  if(motorRpm_ >= 2700 && motorRpm_ <= 3300) poles_ = 2;
20  else if(motorRpm_ >= 1350 && motorRpm_ <= 1650) poles_ = 4;
21  else if(motorRpm_ >= 900 && motorRpm_ <= 1100) poles_ = 6;
22  else if(motorRpm_ >= 675 && motorRpm_ <= 825) poles_ = 8;
23  else if(motorRpm_ >= 540 && motorRpm_ <= 660) poles_ = 10;
24  else if(motorRpm_ >= 450 && motorRpm_ <= 550) poles_ = 12;
25  }
26 
27  /*
28  * Backup case when the Rpm does not fall within any range. Currently the equation for 60 Hz is used. 50 Hz equation
29  * is not available at the time of implementing this.
30  */
31 
32  if (poles_ == 0){
33  if (motorRpm_ > 3600) {
34  poles_ = 2;

36  if (floor(floor(7200 / motorRpm_) / 2) * 2 > 12) {
37  poles_ = 12;

41  }
